<html>
<table>
  <thead>
    <tr>

  </thead>
  <tbody>

          ## H2<br>

    </tr>
    <tr>

      <td><code>**bold text**</code></td>
    </tr>
    <tr>

      <td><code>*italicized text*</code></td>

    <tr>

    <tr>

    </tr>
    <tr>
